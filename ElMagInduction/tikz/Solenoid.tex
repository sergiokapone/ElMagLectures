% Newcommand
\newcommand{\midlabelline}[3]{
   \node (midlabel) at ($ (#1)!.5!(#2) $) {\huge #3};
   \draw[<-, very thick] (#1) --  (midlabel);
   \draw[->|, very thick] (midlabel) -- (#2);
}
\newcommand{\midlabellinee}[3]{
   \node (midlabel) at ($ (#1)!.5!(#2) $) {\huge #3};
   \draw[|<-, very thick] (#1) --  (midlabel);
   \draw[->|, very thick] (midlabel) -- (#2);
}

\newcommand{\midlabellineee}[3]{
   \node (midlabel) at ($ (#1)!.5!(#2) $) {\huge #3};
   \draw[very thick] (#1) --  (midlabel);
   \draw[very thick] (midlabel) -- (#2);
}
